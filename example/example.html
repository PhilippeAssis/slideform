O código a baixo é um exemplo do envio do formulário.
Primeiro ele identifica qual o quadro que vai enviar o fomulário, no nosso caso o "sendForm" depois ele pega todos os valores usando this.result(), envia o formulário (essa é a parte comentada, nesse exemplo ele só exibe os dados enviados no próximo quadro) e passa para o próximo quadro usando this.nextSection() 

var myForm = $("#myForm").slideform();

myForm.beforeNext = function () {
    if (this.current.name == 'sendForm') {
        var result = this.result();
        $('#result').text(result);

        /*
        $.post('submit.php', result, function (data) {
            console.log(data)
        })
        */

        this.nextSection()
    }

    return true;
}
